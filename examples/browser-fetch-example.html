How it works

// In browser via CDN:
import { HLS } from 'https://cdn.jsdelivr.net/npm/streaming-hls-parser/dist/index.browser.js';

// Parse HLS string
const playlist = HLS.parse(hlsString);

// Or fetch from URL (requires CORS)
const response = await fetch(url);
const text = await response.text();
const playlist = HLS.parse(text);

// Or load from file
const file = input.files[0];
const text = await file.text();
const playlist = HLS.parse(text);

// Convert back to HLS format
const hlsString = HLS.stringify(playlist);
